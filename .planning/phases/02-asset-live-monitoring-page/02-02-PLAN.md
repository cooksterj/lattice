**A. State machine (3 states: idle, running, completed/failed):**
```javascript
let currentState = 'idle'; // 'idle' | 'running' | 'completed' | 'failed'
let logCount = 0;
const MAX_LOG_ENTRIES = 2000;

function setState(newState) {
    currentState = newState;
    updateUIForState();
}

function updateUIForState() {
    const logContainer = document.getElementById('log-container');
    const assetInfo = document.getElementById('asset-info');
    const statusText = document.getElementById('status-text');

    switch (currentState) {
        case 'idle':
            logContainer.style.display = 'none';
            assetInfo.style.display = 'block';
            statusText.textContent = 'AWAITING EXECUTION...';
            statusText.className = 'status-text idle';
            break;
        case 'running':
            logContainer.style.display = 'block';
            assetInfo.style.display = 'none';
            statusText.textContent = 'STREAMING LOGS...';
            statusText.className = 'status-text running';
            break;
        case 'completed':
        case 'failed':
            logContainer.style.display = 'block';
            assetInfo.style.display = 'none';
            statusText.textContent = currentState === 'completed' ? 'EXECUTION COMPLETE' : 'EXECUTION FAILED';
            statusText.className = `status-text ${currentState}`;
            break;
    }
}
```

**B. WebSocket connection (following graph.js pattern):**
```javascript
let ws = null;

function connectWebSocket() {
    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
    const url = `${protocol}//${window.location.host}/ws/asset/${ASSET_KEY}`;

    updateConnectionStatus('connecting');
    ws = new WebSocket(url);

    ws.onopen = () => {
        updateConnectionStatus('connected');
    };

    ws.onmessage = (event) => {
        const message = JSON.parse(event.data);
        handleMessage(message);
    };

    ws.onclose = () => {
        updateConnectionStatus('disconnected');
        setTimeout(connectWebSocket, 2000); // Reconnect after 2s
    };

    ws.onerror = () => {
        updateConnectionStatus('error');
    };
}
```

**C. Message handler (handles all 4 Phase 1 message types):**
```javascript
function handleMessage(message) {
    switch (message.type) {
        case 'replay':
            // Replay buffer entries -- process each as if live
            if (message.data && message.data.entries) {
                for (const entry of message.data.entries) {
                    handleMessage(entry);
                }
            }
            break;

        case 'asset_start':
            // New execution starting -- clear logs and switch to running
            clearLogContainer();
            setState('running');
            break;

        case 'asset_log':
            // Log entry -- transition to running if not already, append entry
            if (currentState !== 'running') {
                setState('running');
            }
            appendLogEntry(message.data);
            break;

        case 'asset_complete':
            // Execution finished -- set state based on status
            // Note: message.data.status is "completed" or "failed"
            setState(message.data.status);
            break;
    }
}
```

**D. Log entry rendering (XSS-safe with textContent):**
```javascript
function appendLogEntry(data) {
    const container = document.getElementById('log-container');
    const isAtBottom = container.scrollHeight - container.scrollTop <= container.clientHeight + 50;

    const entry = document.createElement('div');
    entry.className = 'log-entry';

    // Level badge -- controlled string, safe for innerHTML
    const levelSpan = document.createElement('span');
    levelSpan.className = `log-level ${data.level}`;
    levelSpan.textContent = data.level;
    entry.appendChild(levelSpan);

    // Timestamp -- controlled string
    const timeSpan = document.createElement('span');
    timeSpan.className = 'log-timestamp';
    timeSpan.textContent = formatTimestamp(data.timestamp);
    entry.appendChild(timeSpan);

    // Message -- USER-PROVIDED STRING, MUST use textContent for XSS safety
    const msgSpan = document.createElement('span');
    msgSpan.className = 'log-message';
    msgSpan.textContent = data.message;
    entry.appendChild(msgSpan);

    container.appendChild(entry);
    logCount++;

    // Cap DOM entries at MAX_LOG_ENTRIES
    while (logCount > MAX_LOG_ENTRIES && container.firstChild) {
        container.removeChild(container.firstChild);
        logCount--;
    }

    // Auto-scroll only if user was at bottom
    if (isAtBottom) {
        entry.scrollIntoView({ behavior: 'smooth' });
    }
}

function formatTimestamp(isoString) {
    try {
        const d = new Date(isoString);
        return d.toLocaleTimeString('en-US', { hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit' });
    } catch {
        return isoString;
    }
}

function clearLogContainer() {
    const container = document.getElementById('log-container');
    container.innerHTML = '';
    logCount = 0;
}
```

**F. Initial state check via REST (Pitfall 2 from research):**

On page load, BEFORE connecting the WebSocket, check `/api/execution/status` to determine initial state. The replay buffer only contains `asset_log` messages, not `asset_start`. If an execution is already running for this asset, we need to set state to `running` before the WebSocket delivers replayed logs.

```javascript
async function checkInitialState() {
    try {
        const resp = await fetch('/api/execution/status');
        if (!resp.ok) return;
        const status = await resp.json();
        if (status.is_running && status.asset_statuses) {
            const assetStatus = status.asset_statuses.find(a => a.id === ASSET_KEY);
            if (assetStatus) {
                if (assetStatus.status === 'running') {
                    setState('running');
                } else if (assetStatus.status === 'completed' || assetStatus.status === 'failed') {
                    setState(assetStatus.status);
                }
            }
        }
    } catch (e) {
        console.error('Failed to check execution status:', e);
    }
}
```
