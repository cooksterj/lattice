```javascript
function showCompletionBanner(data) {
    const banner = document.getElementById('completion-banner');
    const isSuccess = data.status === 'completed';

    // Reset classes and show
    banner.className = `completion-banner ${isSuccess ? 'success' : 'failure'}`;

    // Build content
    banner.innerHTML = '';

    const icon = document.createElement('span');
    icon.className = 'banner-icon';
    icon.textContent = isSuccess ? '\u2714' : '\u2718';  // checkmark or X
    banner.appendChild(icon);

    const status = document.createElement('span');
    status.className = 'banner-status';
    status.textContent = isSuccess ? 'EXECUTION COMPLETE' : 'EXECUTION FAILED';
    banner.appendChild(status);

    const duration = document.createElement('span');
    duration.className = 'banner-duration';
    duration.textContent = formatDuration(data.duration_ms);
    banner.appendChild(duration);

    // If failed and error message exists, add it
    if (!isSuccess && data.error) {
        const errorDiv = document.createElement('div');
        errorDiv.className = 'banner-error';
        errorDiv.style.cssText = 'width: 100%; font-size: 0.75rem; font-weight: 400; margin-top: 0.5rem; font-family: "Space Mono", monospace;';
        errorDiv.textContent = data.error;  // textContent for safety
        banner.appendChild(errorDiv);
    }
}

// Duration formatter -- same pattern as asset_detail.html and history.html
function formatDuration(ms) {
    if (ms == null) return '';
    if (ms < 1000) return `${Math.round(ms)}ms`;
    if (ms < 60000) return `${(ms / 1000).toFixed(1)}s`;
    const minutes = Math.floor(ms / 60000);
    const seconds = ((ms % 60000) / 1000).toFixed(0);
    return `${minutes}m ${seconds}s`;
}
```

```javascript
function refocusMainWindow() {
    if (window.opener && !window.opener.closed) {
        window.opener.focus();
    } else {
        // Fallback: open graph in new tab (page was opened directly, not via window.open)
        window.open('/', '_blank');
    }
}

function openRunHistory() {
    // Open the asset detail page (run history) in a new browser window
    window.open('/asset/' + ASSET_KEY, '_blank');
}
```
